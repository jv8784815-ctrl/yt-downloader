Descargar de YouTube

Endpoints

POST /api/download

{"url": "...", "format": "mp3"}

Baileys

const API = 'https://TU_API';
const axios = require('axios');
const API = 'https://TU_API';

async function descargarYouTube(sock, from, url) {
    const res = await axios.post(`${API}/api/download`, {url, format: 'mp4'});
    const taskId = res.data.task_id;
    let status;
    do { await new Promise(r => setTimeout(r, 2000)); status = await axios.get(`${API}/api/status/${taskId}`); }
    while (status.data.status !== 'completed');
    await sock.sendMessage(from, { video: { url: `${API}/api/stream/${taskId}` }, mimetype: 'video/mp4' });
}

const API = 'https://TU_API';
const axios = require('axios');
const API = 'https://TU_API';

async function buscarLetra(sock, from, query) {
    const res = await axios.get(`${API}/api/lyrics?q=${encodeURIComponent(query)}`);
    if (res.data.success) {
        await sock.sendMessage(from, { text: `🎵 *${res.data.title}*\n👤 *${res.data.artist}*\n\n${res.data.lyrics.substring(0, 1000)}` });

}

const FormData = require('form-data');
const API = 'https://TU_API';

async function subirArchivo(sock, from, fileBuffer, filename) {
    const form = new FormData();
    form.append('file', fileBuffer, filename);
    form.append('compress', 'true');
    const res = await axios.post(`${API}/api/hosting/upload`, form, { headers: form.getHeaders() });
    await sock.sendMessage(from, { text: `✅ *Archivo subido*\n📎 Link: ${API}/api/hosting/view/${res.data.file_id}` });
}

const API = 'https://TU_API';
const axios = require('axios');
const FormData = require('form-data');
const API = 'https://TU_API';

async function convertirAGif(sock, from, fileBuffer, filename) {
    const form = new FormData();
    form.append('file', fileBuffer, filename);
    form.append('width', '480');
    form.append('fps', '10');
    const res = await axios.post(`${API}/api/gif/convert`, form, { headers: form.getHeaders() });
    await sock.sendMessage(from, { 
        video: { url: `${API}/api/hosting/view/${res.data.file_id}` }, 
        gifPlayback: true,
        caption: '🎞️ GIF generado' 
    });
}
