Baileys Integration

const axios = require('axios');
const API = 'https://yt-downloader-by-josexo-and-abejitadiceiwiwiwi.vercel.app';

async function download(sock, from, url) {
    const res = await axios.post(`${API}/api/download`, {
        url, format: 'mp4', quality: 'medium'
    });
    const taskId = res.data.task_id;
    let status;
    do {
        await new Promise(r => setTimeout(r, 2000));
        status = await axios.get(`${API}/api/status/${taskId}`);
    } while (status.data.status !== 'completed');
    await sock.sendMessage(from, {
        video: { url: `${API}/api/stream/${taskId}` },
        mimetype: 'video/mp4'
    });
}
