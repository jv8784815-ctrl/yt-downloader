const axios = require('axios');
const API = 'https://yt-downloader-by-josexo-and-abejitadiceiwiwiwi.vercel.app';

async function downloadYT(sock, from, url) {
    const res = await axios.post(`${API}/api/download`, {url, format: 'mp4', quality: 'medium'});
    const taskId = res.data.task_id;
    let status;
    do { await new Promise(r => setTimeout(r, 2000)); status = await axios.get(`${API}/api/status/${taskId}`); }
    while (status.data.status !== 'completed');
    await sock.sendMessage(from, { video: { url: `${API}/api/stream/${taskId}` }, mimetype: 'video/mp4', caption: status.data.title });
}

Lyrics + Baileys

const API = 'https://yt-downloader-by-josexo-and-abejitadiceiwiwiwi.vercel.app';
const axios = require('axios');
const API = 'https://yt-downloader-by-josexo-and-abejitadiceiwiwiwi.vercel.app';

async function getLyrics(sock, from, query) {
    const res = await axios.get(`${API}/api/lyrics?q=${encodeURIComponent(query)}`);
    const data = res.data;
    if (data.success) {
        await sock.sendMessage(from, { text: `🎵 *${data.title}*\n👤 *${data.artist}*\n\n${data.lyrics.substring(0, 1000)}` });
    }
}
