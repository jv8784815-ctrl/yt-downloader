Baileys Integration

const axios = require('axios');
const API = 'https://yt-downloader-by-josexo-and-abejitadiceiwiwiwi.vercel.app';

async function download(sock, from, url) {
    const res = await axios.post(`${API}/api/download`, {
        url, format: 'mp4', quality: 'medium'
    });
    
    const taskId = res.data.task_id;
    
    let status;
    do {
        await new Promise(r => setTimeout(r, 2000));
        status = await axios.get(`${API}/api/status/${taskId}`);
    } while (status.data.status !== 'completed');
    
    await sock.sendMessage(from, {
        video: { url: `${API}/api/stream/${taskId}` },
        mimetype: 'video/mp4',
        caption: status.data.title
    });
}

Baileys Integration - Lyrics

Command: .lyrics

const axios = require('axios');
const API = 'https://yt-downloader-by-josexo-and-abejitadiceiwiwiwi.vercel.app';

// Comando: .lyrics Fleetwood Mac - The Chain
async function lyricsCommand(sock, from, query) {
    try {
        const res = await axios.get(`${API}/api/lyrics?q=${encodeURIComponent(query)}`);
        const data = res.data;
        
        if (data.success) {
            let message = `🎵 *${data.title}*\n👤 *${data.artist}*\n\n`;
            
            let lyrics = data.lyrics;
            if (lyrics.length > 1000) {
                lyrics = lyrics.substring(0, 1000) + '...';
            }
            
            message += lyrics;
            
            await sock.sendMessage(from, { text: message });
        } else {
            await sock.sendMessage(from, { 
                text: `❌ No lyrics found for: ${query}` 
            });
        }
    } catch (error) {
        await sock.sendMessage(from, { 
            text: '❌ Error fetching lyrics' 
        });
    }
}

// En tu handler:
if (text.startsWith('.lyrics ')) {
    const query = text.replace('.lyrics ', '');
    await lyricsCommand(sock, from, query);
}
