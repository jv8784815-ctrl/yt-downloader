📚 Documentación de la API

POST /api/download

Inicia una descarga de audio o video

{
  "url": "https://youtu.be/...",
  "format": "mp3",
  "quality": "best"
}

GET /api/status/{task_id}

Verifica el estado de una descarga

GET /api/download/{task_id}

Descarga el archivo procesado

GET /api/info?url=URL

Obtiene información sin descargar

🤖 Integración con Baileys (WhatsApp Bot)

📦 Instalación

npm install baileys axios

🔧 Código para tu bot

const axios = require('axios');

const API_URL = 'http://64.20.54.50:30325';

async function descargarMusica(url) {
    const response = await axios.post(`${API_URL}/api/download`, {
        url: url,
        format: 'mp3',
        quality: 'best'
    });
    
    const taskId = response.data.task_id;
    
    let status;
    do {
        await new Promise(r => setTimeout(r, 2000));
        status = await axios.get(`${API_URL}/api/status/${taskId}`);
    } while (status.data.status !== 'completed');
    
    return `${API_URL}${status.data.download_url}`;
}

🎯 Comandos sugeridos
