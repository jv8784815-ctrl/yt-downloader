CODE

const axios = require('axios');

const API_URL = 'https://your-vercel.vercel.app';

async function downloadAndSend(sock, from, url) {
    // 1. Start download
    const res = await axios.post(`${API_URL}/api/download`, {
        url: url,
        format: 'mp4',
        quality: 'medium'
    });
    
    const taskId = res.data.task_id;
    
    // 2. Wait for completion
    let status;
    do {
        await new Promise(r => setTimeout(r, 2000));
        status = await axios.get(`${API_URL}/api/status/${taskId}`);
    } while (status.data.status !== 'completed');
    
    // 3. Send directly (no local download)
    await sock.sendMessage(from, {
        video: { url: `${API_URL}/api/stream/${taskId}` },
        mimetype: 'video/mp4',
        caption: status.data.title
    });
}
