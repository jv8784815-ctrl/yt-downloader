Descargar de YouTube

Endpoints de FrikiTube

POST /api/download

Inicia una descarga de YouTube

{
  "url": "https://youtu.be/...",
  "format": "mp3",
  "quality": "medium"
}

GET /api/status/{task_id}

Verifica el estado de la descarga

GET /api/stream/{task_id}

Descarga el archivo procesado

GET /api/info?url=...

Obtiene información del video

Integración con Baileys

const API = 'https://yt-downloader-by-josexo-and-abejitadiceiwiwiwi.vercel.app';
const axios = require('axios');
const API = 'https://yt-downloader-by-josexo-and-abejitadiceiwiwiwi.vercel.app';

async function descargarYouTube(sock, from, url) {
    const res = await axios.post(`${API}/api/download`, {
        url, format: 'mp4', quality: 'medium'
    });
    const taskId = res.data.task_id;
    let status;
    do {
        await new Promise(r => setTimeout(r, 2000));
        status = await axios.get(`${API}/api/status/${taskId}`);
    } while (status.data.status !== 'completed');
    
    await sock.sendMessage(from, {
        video: { url: `${API}/api/stream/${taskId}` },
        mimetype: 'video/mp4',
        caption: status.data.title
    });
}

// Buscar y enviar TikTok aleatorio
async function tiktokAleatorio(sock, from, query) {
    const random = await axios.get(`${API}/api/tiktok/random?q=${encodeURIComponent(query)}`);
    if (random.data.success) {
        await descargarTikTok(sock, from, random.data.url);
    }
}

Buscar Letra de Canción

Fleetwood Mac Bad Bunny

Endpoints de FrikiLyric

GET /api/lyrics?q=...

Busca la letra de una canción

{
  "success": true,
  "artist": "Fleetwood Mac",
  "title": "The Chain",
  "lyrics": "Listen to the wind blow...",
  "source": "lrclib"
}

Integración con Baileys

const API = 'https://yt-downloader-by-josexo-and-abejitadiceiwiwiwi.vercel.app';
const axios = require('axios');
const API = 'https://yt-downloader-by-josexo-and-abejitadiceiwiwiwi.vercel.app';

async function buscarLetra(sock, from, query) {
    const res = await axios.get(`${API}/api/lyrics?q=${encodeURIComponent(query)}`);
    const data = res.data;
    
    if (data.success) {
        let mensaje = `🎵 *${data.title}*\n👤 *${data.artist}*\n\n`;
        mensaje += data.lyrics.substring(0, 1000);
        if (data.lyrics.length > 1000) mensaje += '...';
        
        await sock.sendMessage(from, { text: mensaje });
    } else {
        await sock.sendMessage(from, { text: '❌ No se encontró la letra' });
    }
}

// Comando: .letra Fleetwood Mac - The Chain
